 *  INPUT â†’ LANGUAGE MAPPING (Region-specific)
 ************************************************************/

Object.assign(Texts_LangMenus, {

  /************************************************************
   * Map numeric choice â†’ language code
   * region = "PK" | "IN" | "BD" | "ME" | "OTHER"
   ************************************************************/
  mapChoiceToLanguage(region, choice) {
    choice = String(choice || "").trim();

    switch (region) {

      /********************* ğŸ‡µğŸ‡° PK REGION *********************/
      case "PK":
        // Valid: 1 = Urdu, 2 = Roman Urdu, 3 = English
        if (choice === "1") return "URDU";
        if (choice === "2") return "ROMAN_URDU";
        if (choice === "3") return "ENGLISH";
        return null;

      /********************* ğŸ‡®ğŸ‡³ IN REGION *********************/
      case "IN":
        // Valid: 1 = Hindi, 2 = English
        if (choice === "1") return "HINDI";
        if (choice === "2") return "ENGLISH";
        return null;

      /********************* ğŸ‡§ğŸ‡© BD REGION *********************/
      case "BD":
        // Valid: 1 = Bangla, 2 = English
        if (choice === "1") return "BANGLA";
        if (choice === "2") return "ENGLISH";
        return null;

      /********************* ğŸ‡¸ğŸ‡¦ ME REGION *********************/
      case "ME":
        // Valid: 1 = Arabic, 2 = English
        if (choice === "1") return "ARABIC";
        if (choice === "2") return "ENGLISH";
        return null;

      /********************* ğŸŒ OTHER REGION *********************/
      case "OTHER":
      default:
        // Valid: 1â€“6 (full global list)
        if (choice === "1") return "ENGLISH";
        if (choice === "2") return "URDU";
        if (choice === "3") return "ROMAN_URDU";
        if (choice === "4") return "HINDI";
        if (choice === "5") return "BANGLA";
        if (choice === "6") return "ARABIC";
        return null;
    }
  },

  /************************************************************
   * Validate if input is allowed for this region
   ************************************************************/
  isValidLanguageChoice(region, choice) {
    choice = String(choice || "").trim();

    if (region === "PK")   return ["1", "2", "3"].includes(choice);
    if (region === "IN")   return ["1", "2"].includes(choice);
    if (region === "BD")   return ["1", "2"].includes(choice);
    if (region === "ME")   return ["1", "2"].includes(choice);
    if (region === "OTHER")return ["1", "2", "3", "4", "5", "6"].includes(choice);

    return false;
  },

  /************************************************************
   * Return the correct menu object for the region
   ************************************************************/
  getMenuForRegion(region) {
    switch (region) {
      case "PK":    return this._MENU_PK;
      case "IN":    return this._MENU_IN;
      case "BD":    return this._MENU_BD;
      case "ME":    return this._MENU_ME;
      case "OTHER": return this._MENU_OTHER;
      default:      return this._MENU_OTHER; // safety fallback
    }
  },

  /************************************************************
   * Helper to send menu message in userâ€™s own selected language
   ************************************************************/
  getMenuText(region, userLang) {
    const menu = this.getMenuForRegion(region);
    return menu[userLang] || menu["English"]; // fallback
  }

});
