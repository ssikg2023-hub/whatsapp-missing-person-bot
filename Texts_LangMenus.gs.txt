/************************************************************
 *  Texts_LangMenus.gs ‚Äî BLOCK 3/3
 *  INPUT ‚Üí LANGUAGE MAPPING (Region-specific)
 ************************************************************/

Object.assign(Texts_LangMenus, {

  /************************************************************
   * Map numeric choice ‚Üí language code
   * region = "PK" | "IN" | "BD" | "ME" | "OTHER"
   ************************************************************/
  mapChoiceToLanguage(region, choice) {
    const options = this.getLanguageOptions(region);
    const trimmed = String(choice || "").trim();
    const mapped = options[trimmed] || null;

    const normalize = {
      ENGLISH: "English",
      URDU: "Urdu",
      ROMAN_URDU: "RomanUrdu",
      HINDI: "Hindi",
      BANGLA: "Bangla",
      ARABIC: "Arabic",
    };

    if (!mapped) return null;
    return normalize[mapped] || normalize[(mapped || "").toUpperCase()] || mapped;
  },

  /************************************************************
   * Alias used by Flows.gs routing
   ************************************************************/
  mapChoiceToLang(region, choice) {
    return this.mapChoiceToLanguage(region, choice);
  },

  /************************************************************
   * Validate if input is allowed for this region
   ************************************************************/
  isValidLanguageChoice(region, choice) {
    const allowed = this.getAllowedOptions(region);
    const trimmed = String(choice || "").trim();
    return allowed.includes(trimmed);
  },

  /************************************************************
   * Return the correct menu object for the region
   ************************************************************/
  getMenuForRegion(region) {
    switch (region) {
      case "PK":    return this._MENU_PK;
      case "IN":    return this._MENU_IN;
      case "BD":    return this._MENU_BD;
      case "ME":    return this._MENU_ME;
      case "OTHER": return this._MENU_OTHER;
      default:      return this._MENU_OTHER; // safety fallback
    }
  },

  /************************************************************
   * Helper to send menu message in user‚Äôs own selected language
   ************************************************************/
  getMenuText(region, userLang) {
    const menu = this.getMenuForRegion(region);
    return menu[userLang] || menu["English"]; // fallback
  }

});
